00075     int evaluate(Request* req){
00076         int result = std::fabs(req->getDeparture()-curFloor)+req->getLength();
00077         if(curDirection != Request::DIRECTION_NONE){
00078             Request::DIRECTION dirReq = req->getDirectionToDeparture(curFloor);
00079             if(dirReq!=Request::DIRECTION_NONE&&curDirection != dirReq)
00080                 return REQUEST_UNACCEPTABLE;
00081             int pole = getPoleDepartureInCurrentDirection();
00082             if(direction != req->getDirection()){
00083                 if(curDirection==direction) return REQUEST_UNACCEPTABLE;
00084                 if(curDirection==Request::DIRECTION_UP){
00085                     if(pole < req->getDeparture() || pole < req->getDestination()) return REQUEST_UNACCEPTABLE;
00086                 }else if(curDirection==Request::DIRECTION_DOWN){
00087                     if(pole > req->getDeparture() || pole > req->getDestination()) return REQUEST_UNACCEPTABLE;
00088                 }
00089                 result+=2*std::fabs(req->getDeparture()-pole);
00090             }else{
00091                 if(curDirection!=direction)
00092                     return REQUEST_UNACCEPTABLE;
00093             }
00094         }
00095         return result;
00096     }

00369     void nextFloor(){
00370         bool hasDoorOpened = false;
00371         bool hasDeparture = false;
00372         /*
00373             if has departure
00374                 open door
00375                 unsetDeparture
00376                 query whether has departure if not change curDirection
00377           */
00378         inPassengerCount = outPassengerCount = 0;
00379         for(int i = requests.count()-1; i >= 0; i--){
00380             if(curFloor==requests.at(i)->getDeparture()){
00381                 requests.at(i)->unsetDeparture();
00382                 passengerCount++;
00383                 inPassengerCount++;
00384                 hasDoorOpened=true;
00385             }else{
00386                 if(requests.at(i)->getDeparture()!=FLOOR_NONE){
00387                     hasDeparture=true;
00388                 }else if(curFloor==requests.at(i)->getDestination()){
00389                     requests.removeAt(i);
00390                     passengerCount--;
00391                     outPassengerCount++;
00392                     hasDoorOpened=true;
00393                 }
00394             }
00395         }
00396 
00397         if(hasDoorOpened){
00398             if(curDirection!=direction&&hasDeparture==false){
00399                 setCurDirection(direction);
00400                 emit signalCanAcceptRequest(id);
00401             }
00402             action();
00403             return;
00404         }
00405 
00406         if(isIdle()){
00407             direction=Request::DIRECTION_NONE;
00408             setCurDirection(direction);
00409             emit signalCanAcceptRequest(id);
00410             return;
00411         }
00412 
00413         if(curDirection==Request::DIRECTION_UP)
00414             curFloor++;
00415         else if(curDirection==Request::DIRECTION_DOWN)
00416             curFloor--;
